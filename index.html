Q-1 You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to remove all the duplicate occurrences of characters in the string. Input: abcadeecfb Output: abcdef" 
ans1:-
               function removeDuplicates(inputStr) {
                const charArray = inputStr.split(''); 
                const seenChars = new Set();
                
                const result = charArray.map(char => {
                  if (!seenChars.has(char)) {
                    seenChars.add(char);
                    return char;
                  }
                }).join(''); 
                
                return result;
              }
